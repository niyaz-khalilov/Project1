CREATE TABLE Person (
    personId int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    fullName varchar(100) NOT NULL UNIQUE,
    birthdayDate int NOT NULL
);

CREATE TABLE Book (
    bookId int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    tittle varchar(100) NOT NULL,
    author varchar(100) NOT NULL,
    issueyear int NOT NULL,
    personId int REFERENCES Person(personId) ON DELETE SET NULL
);
